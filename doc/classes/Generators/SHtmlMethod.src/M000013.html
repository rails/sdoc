    def github_url
      if @source_code =~ /File\s(\S+), line (\d+)/
        file = $1
        line = $2.to_i
      end
      url = SHTMLGenerator.github_url(file, @options)
      unless line.nil? || url.nil?
        url + "#L#{line}" 
      else
        ''
      end
    end
